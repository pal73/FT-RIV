Re: FW: FT813

Павел, добрый день


Если мне память не изменяет, то вы STM32 используете в данном проекте? Если так, то самый простой путь - взять готовый пример от Riverdi под этот контроллер (ссылка на архив):

http://mymcu.ru/storage/content/files/Riverdi/Revelation_Board_FT81X_demo_firmware_source_Rev.1.2.zip


Он базируется на оф. примере от FTDI, чуть изменена часть, касающаяся инициализации. Есть еще нюанс во всех модулях Riverdi с FT81x - на платах не ставится внешний генератор, используется внутренний (в примерах от FTDI используется внешний, т.к на всех отладках FTDI внешний генератор для FT8xx установлен). Этот нюанс в документации Riverdi в явном виде не отражен, и вызывает иногда проблемы.

В этом примере найдете все параметры для 7", код 100% рабочий - я в своих тестах его использовал. По структуре проекта - если за основу брать его, то потом можно экспортом из Screen Editor пользоваться, что удобно на начальных этапах, меньше вероятность ошибки в порядке команд, работы с изображениями возникает. И я смогу такой код у себя проверить, если понадобиться. 



сам процесс инициализации в более читаемом виде, чем в main() в архиве. Привожу для лучшего понимания. Работают оба варианта :) 


ft_void_t SAMAPP_BootupConfig()

{

       /* Do a power cycle for safer side */

       Ft_Gpu_Hal_Powercycle(phost,FT_TRUE); // в этой функции PD передергиваем


       // следующий набор команд отвечает за включение тактового генератора и запуск контроллера в раб. состояние - команды имеют названия, как в руководстве программиста 


       /* Access address 0 to wake up the FT800 */

       Ft_Gpu_HostCommand(phost,FT_GPU_ACTIVE_M);  

       Ft_Gpu_Hal_Sleep(20);


       /* Set the clk to external clock */

       Ft_Gpu_HostCommand(phost,FT_GPU_INTERNAL_OSC);

       Ft_Gpu_Hal_Sleep(10);

         


       Ft_Gpu_HostCommand(phost,FT_GPU_SLEEP_M);


       Ft_Gpu_81X_SelectSysCLK(phost,FT_GPU_SYSCLK_48M);


       Ft_Gpu_HostCommand(phost,FT_GPU_ACTIVE_M);


       /* Do a core reset for safer side */

       Ft_Gpu_HostCommand(phost,FT_GPU_CORE_RESET);


// далее чтением регистра проверяем, что контроллер запустился, в принципе читать можно любой регистр - этот удобен тем, что имеет неизменяемое значение


       {

               ft_uint8_t chipid;

               //Read Register ID to check if FT800 is ready. 

               chipid = Ft_Gpu_Hal_Rd8(phost, REG_ID);

               while(chipid != 0x7C)

                       chipid = Ft_Gpu_Hal_Rd8(phost, REG_ID);

// загрузка параметров дисплея


    Ft_Gpu_Hal_Wr16(phost, REG_HCYCLE, FT_DispHCycle);

    Ft_Gpu_Hal_Wr16(phost, REG_HOFFSET, FT_DispHOffset);

    Ft_Gpu_Hal_Wr16(phost, REG_HSYNC0, FT_DispHSync0);

    Ft_Gpu_Hal_Wr16(phost, REG_HSYNC1, FT_DispHSync1);

    Ft_Gpu_Hal_Wr16(phost, REG_VCYCLE, FT_DispVCycle);

    Ft_Gpu_Hal_Wr16(phost, REG_VOFFSET, FT_DispVOffset);

    Ft_Gpu_Hal_Wr16(phost, REG_VSYNC0, FT_DispVSync0);

    Ft_Gpu_Hal_Wr16(phost, REG_VSYNC1, FT_DispVSync1);

    Ft_Gpu_Hal_Wr8(phost, REG_SWIZZLE, FT_DispSwizzle);

    Ft_Gpu_Hal_Wr8(phost, REG_PCLK_POL, FT_DispPCLKPol);

    Ft_Gpu_Hal_Wr16(phost, REG_HSIZE, FT_DispWidth);

    Ft_Gpu_Hal_Wr16(phost, REG_VSIZE, FT_DispHeight);

    Ft_Gpu_Hal_Wr16(phost, REG_CSPREAD, FT_DispCSpread);

    Ft_Gpu_Hal_Wr16(phost, REG_DITHER, FT_DispDither);



    /* Touch configuration - configure the resistance value to 1200 - this value is specific to customer requirement and derived by experiment */

    Ft_Gpu_Hal_Wr16(phost, REG_TOUCH_RZTHRESH,1200);


// инициализация портов ввода/вывода

 

    Ft_Gpu_Hal_Wr8(phost, REG_GPIO_DIR,0xff);

    Ft_Gpu_Hal_Wr8(phost, REG_GPIO,0xff);

// простенький код - меняет цвет экрана, код ниже - на низком уровне реализован простой дисплей-лист 

    Ft_Gpu_Hal_WrMem(phost, RAM_DL,(ft_uint8_t *)FT_DLCODE_BOOTUP,sizeof(FT_DLCODE_BOOTUP));

    Ft_Gpu_Hal_Wr8(phost, REG_DLSWAP,DLSWAP_FRAME);


// подача рабочей частоты на дисплей - будет видно изображение, загруженное двумя предыдущими командами 

    Ft_Gpu_Hal_Wr8(phost, REG_PCLK,FT_DispPCLK);//after this display is visible on the LCD



}



const ft_uint8_t FT_DLCODE_BOOTUP[12] = 

{

    0,0,0,2,//GPU instruction CLEAR_COLOR_RGB

    7,0,0,38, //GPU instruction CLEAR

    0,0,0,0,  //GPU instruction DISPLAY

};
